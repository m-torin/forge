# Better Auth

Forge uses [Better Auth](https://better-auth.com) for authentication. Better Auth is a
comprehensive, open-source authentication framework for TypeScript that integrates seamlessly with
Next.js and provides many features out of the box.

## Getting Started

### Installation

First, install Better Auth dependencies in your auth package:

```bash
pnpm add better-auth better-auth@next @repo/database --filter @repo/auth
```

### Environment Variables

Set up the required environment variables in your `.env.local` file:

```env
# Generate with: npx @better-auth/cli secret
BETTER_AUTH_SECRET="your-secret-here"
DATABASE_URL="postgresql://..."
NEXT_PUBLIC_APP_URL="http://localhost:3000"
```

### Server Actions

Use the `auth.api` object to access authentication in server actions:

```typescript title="app/actions/protected.ts"
'use server';

import { auth } from '@repo/auth/server';
import { headers } from 'next/headers';

export async function myServerAction() {
  const session = await auth.api.getSession({
    headers: await headers(),
  });

  if (!session?.user) {
    throw new Error('Unauthorized');
  }

  // Your protected logic here
  return { userId: session.user.id };
}
```

### Next.js 15.2.0+ Middleware with Node.js Runtime

For Next.js 15.2.0 and above, you can use the Node.js runtime in middleware for improved
performance:

```typescript title="app/middleware.ts"
import { nodeAuthMiddleware } from '@repo/auth/middleware';
import { noseconeMiddleware, noseconeOptions } from '@repo/security/middleware';
import type { NextMiddleware, NextRequest } from 'next/server';

const securityHeaders = noseconeMiddleware(noseconeOptions);

export default async function middleware(request: NextRequest) {
  // Use modern Node.js runtime middleware
  const authResponse = await nodeAuthMiddleware(request);

  // If auth middleware returned a response (redirect), use it
  if (authResponse.status !== 200) {
    return authResponse;
  }

  // Otherwise, apply security headers
  return securityHeaders();
}

export const config = {
  runtime: 'nodejs', // Enable Node.js runtime for middleware
  matcher: [
    '/((?!_next|[^?]*\\.(?:html?|css|js(?!on)|jpe?g|webp|png|gif|svg|ttf|woff2?|ico|csv|docx?|xlsx?|zip|webmanifest)).*)',
    '/(api|trpc)(.*)',
  ],
};
```

### Auth Package Middleware

The auth package provides middleware for route protection:

```typescript title="packages/auth/middleware.ts"
import type { NextRequest } from 'next/server';
import { NextResponse } from 'next/server';
import { getSessionCookie } from 'better-auth/cookies';
import { headers } from 'next/headers';
import { auth } from './server';

const publicRoutes = ['/sign-in', '/sign-up', '/api/auth', '/_next', '/favicon.ico'];

const isPublicRoute = (pathname: string) => {
  return publicRoutes.some((route) => pathname.startsWith(route));
};

// Legacy middleware for edge runtime (cookie-only)
export const authMiddleware = (next?: () => NextResponse) => {
  return async (request: NextRequest) => {
    const { pathname } = request.nextUrl;

    if (isPublicRoute(pathname)) {
      return next ? next() : NextResponse.next();
    }

    const sessionCookie = getSessionCookie(request);

    if (!sessionCookie) {
      const signInUrl = new URL('/sign-in', request.url);
      signInUrl.searchParams.set('callbackUrl', pathname);
      return NextResponse.redirect(signInUrl);
    }

    return next ? next() : NextResponse.next();
  };
};

// Modern middleware for Next.js 15.2.0+ with Node.js runtime
export async function nodeAuthMiddleware(request: NextRequest) {
  const { pathname } = request.nextUrl;

  // Skip authentication for public routes
  if (isPublicRoute(pathname)) {
    return NextResponse.next();
  }

  // Check session using auth.api (requires Node.js runtime)
  const session = await auth.api.getSession({
    headers: await headers(),
  });

  if (!session) {
    const signInUrl = new URL('/sign-in', request.url);
    signInUrl.searchParams.set('callbackUrl', pathname);
    return NextResponse.redirect(signInUrl);
  }

  return NextResponse.next();
}
```

### Database User Queries

Access users directly from the database:

```typescript title="app/actions/users/get.ts"
'use server';

import { database } from '@repo/database';
import { getSession } from '@repo/auth/server';

export const getUsers = async (userIds: string[]) => {
  const session = await getSession();

  if (!session?.user) {
    throw new Error('Not logged in');
  }

  // Fetch users from database
  const users = await database.user.findMany({
    where: {
      id: {
        in: userIds,
      },
    },
  });

  return users;
};
```

## Database Schema

Better Auth generates the following database schema:

```prisma
model User {
  id            String    @id
  name          String
  email         String
  emailVerified Boolean
  image         String?
  createdAt     DateTime
  updatedAt     DateTime
  sessions      Session[]
  accounts      Account[]

  @@unique([email])
  @@index([email]) // Performance optimization
  @@map("user")
}

model Session {
  id        String   @id
  expiresAt DateTime
  token     String
  createdAt DateTime
  updatedAt DateTime
  ipAddress String?
  userAgent String?
  userId    String
  user      User     @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@unique([token])
  @@index([userId]) // Performance optimization
  @@index([token]) // Performance optimization
  @@map("session")
}

model Account {
  id                    String    @id
  accountId             String
  providerId            String
  userId                String
  user                  User      @relation(fields: [userId], references: [id], onDelete: Cascade)
  accessToken           String?
  refreshToken          String?
  idToken               String?
  accessTokenExpiresAt  DateTime?
  refreshTokenExpiresAt DateTime?
  scope                 String?
  password              String?
  createdAt             DateTime
  updatedAt             DateTime

  @@index([userId]) // Performance optimization
  @@map("account")
}

model Verification {
  id         String    @id
  identifier String
  value      String
  expiresAt  DateTime
  createdAt  DateTime?
  updatedAt  DateTime?

  @@index([identifier]) // Performance optimization
  @@map("verification")
}
```

#### Phone Number Authentication

```typescript title="Phone authentication"
'use client';

import { signIn } from '@repo/auth/client';
import { useState } from 'react';

export function PhoneSignIn() {
  const [phone, setPhone] = useState('');
  const [code, setCode] = useState('');
  const [codeSent, setCodeSent] = useState(false);

  const sendCode = async () => {
    await signIn.phone.sendCode({ phone });
    setCodeSent(true);
  };

  const verifyCode = async (e: React.FormEvent) => {
    e.preventDefault();
    await signIn.phone.verify({ phone, code });
  };

  if (!codeSent) {
    return (
      <div>
        <input
          type="tel"
          value={phone}
          onChange={(e) => setPhone(e.target.value)}
          placeholder="+1234567890"
        />
        <button onClick={sendCode}>Send code</button>
      </div>
    );
  }

  return (
    <form onSubmit={verifyCode}>
      <input
        type="text"
        value={code}
        onChange={(e) => setCode(e.target.value)}
        placeholder="Enter verification code"
      />
      <button type="submit">Verify</button>
    </form>
  );
}
```

### API Key Authentication

Support API key authentication for machine-to-machine communication:

```typescript title="API key authentication"
import { apiKey } from 'better-auth/plugins';

export const auth = betterAuth({
  plugins: [
    apiKey({
      generateKey: () => {
        // Generate a secure random key
        return crypto.randomBytes(32).toString('hex');
      },
    }),
  ],
});

// Usage in API routes
export async function GET(request: Request) {
  const apiKey = request.headers.get('x-api-key');

  const session = await auth.api.getSession({
    headers: request.headers,
  });

  if (!session && !apiKey) {
    return new Response('Unauthorized', { status: 401 });
  }

  // Validate API key if provided
  if (apiKey) {
    const validKey = await auth.api.validateAPIKey(apiKey);
    if (!validKey) {
      return new Response('Invalid API key', { status: 401 });
    }
  }

  return Response.json({ data: 'Protected data' });
}
```

## Performance Optimizations

Better Auth includes several performance optimizations to ensure fast authentication operations:

### Cookie Cache

The auth package is configured with cookie caching to reduce database queries:

```typescript title="Cookie caching configuration"
export const auth = betterAuth({
  // ... other config
  session: {
    cookieCache: {
      enabled: true,
      maxAge: 5 * 60, // Cache session for 5 minutes
    },
  },
});
```

This stores session data in a short-lived, signed cookie similar to JWT access tokens, reducing the
need for database lookups on every request.

### Database Indexing

The database schema includes performance-optimized indexes on frequently queried fields:

- `User` table: Index on `email` field
- `Session` table: Indexes on `userId` and `token` fields
- `Account` table: Index on `userId` field
- `Verification` table: Index on `identifier` field

These indexes significantly improve query performance for common authentication operations.

### Middleware Optimization

For Next.js 15.2.0+, you can use the Node.js runtime middleware which allows direct database access
without edge limitations:

```typescript title="Node.js runtime configuration"
export const config = {
  runtime: 'nodejs', // Enable Node.js runtime for middleware
  matcher: [
    /* your routes */
  ],
};
```

This enables using `auth.api.getSession()` directly in middleware for more accurate session
verification.

### Efficient Client-Side State

Better Auth uses optimized client-side state management with automatic session refresh, preventing
unnecessary API calls while maintaining security.

## Security Features

Better Auth includes comprehensive security features:

- **Secure Session Management**: httpOnly cookies with SameSite protection
- **CSRF Protection**: Built-in CSRF token validation
- **Rate Limiting**: Configure per-route rate limits
- **Session Fixation Protection**: New session ID after authentication
- **Secure Password Storage**: Bcrypt with configurable rounds
- **Content Security Policy**: Compatible with strict CSP headers
- **Timing Attack Prevention**: Constant-time comparison for secrets

### Custom Authentication Form with Error Handling

```typescript title="Custom auth form"
'use client';

import { signIn, signUp } from '@repo/auth/client';
import { useState } from 'react';
import { useRouter } from 'next/navigation';

export function CustomAuthForm() {
  const [mode, setMode] = useState<'signin' | 'signup'>('signin');
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  const [name, setName] = useState('');
  const [error, setError] = useState<string | null>(null);
  const [loading, setLoading] = useState(false);
  const router = useRouter();

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    setError(null);
    setLoading(true);

    try {
      if (mode === 'signin') {
        await signIn.email({ email, password });
      } else {
        await signUp.email({ email, password, name });
      }
      router.push('/dashboard');
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Authentication failed');
    } finally {
      setLoading(false);
    }
  };

  return (
    <form onSubmit={handleSubmit} className="space-y-4">
      {mode === 'signup' && (
        <input
          type="text"
          value={name}
          onChange={(e) => setName(e.target.value)}
          placeholder="Full name"
          required
          className="w-full px-3 py-2 border rounded"
        />
      )}

      <input
        type="email"
        value={email}
        onChange={(e) => setEmail(e.target.value)}
        placeholder="Email"
        required
        className="w-full px-3 py-2 border rounded"
      />

      <input
        type="password"
        value={password}
        onChange={(e) => setPassword(e.target.value)}
        placeholder="Password"
        required
        className="w-full px-3 py-2 border rounded"
      />

      {error && (
        <div className="text-red-600 text-sm">{error}</div>
      )}

      <button
        type="submit"
        disabled={loading}
        className="w-full bg-blue-600 text-white py-2 rounded hover:bg-blue-700 disabled:opacity-50"
      >
        {loading ? 'Loading...' : mode === 'signin' ? 'Sign In' : 'Sign Up'}
      </button>

      <button
        type="button"
        onClick={() => setMode(mode === 'signin' ? 'signup' : 'signin')}
        className="w-full text-blue-600 hover:underline"
      >
        {mode === 'signin' ? 'Need an account? Sign up' : 'Have an account? Sign in'}
      </button>
    </form>
  );
}
```

## Best Practices

1. **Environment Variables**: Use `BETTER_AUTH_SECRET` with a cryptographically secure value
2. **Database Security**: Use parameterized queries (handled by Prisma adapter)
3. **HTTPS Only**: Always use HTTPS in production
4. **Session Configuration**: Set appropriate expiration times
5. **Error Handling**: Don't expose sensitive information in error messages
6. **Logging**: Log authentication events for security monitoring
7. **Input Validation**: Validate all user inputs before processing
8. **Regular Updates**: Keep Better Auth and dependencies up to date
9. **Component Reusability**: Use the pre-built components from `@repo/auth` when possible
10. **State Management**: Let Better Auth handle auth state internally
