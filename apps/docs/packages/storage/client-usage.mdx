# Client-Side Storage Usage

Implement secure client-side file uploads with progress tracking, error
handling, and seamless UI integration.

## Overview

The storage package provides client-side utilities for:

- Secure uploads without exposing server credentials
- Real-time progress tracking
- Multipart upload coordination
- Error handling and retry logic
- React component integration

## Basic File Upload

### Simple Upload with Presigned URL

```typescript
'use client';

import { useState } from 'react';
import { uploadWithPresignedUrl } from '@repo/storage/client/next';
import { logInfo, logError } from '@repo/observability/server/next';

export function SimpleUploader() {
  const [uploading, setUploading] = useState(false);
  const [progress, setProgress] = useState(0);

  const handleUpload = async (file: File) => {
    setUploading(true);
    setProgress(0);

    try {
      // Get presigned URL from your API
      const response = await fetch('/api/upload-url', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          filename: file.name,
          contentType: file.type,
          size: file.size,
        }),
      });

      if (!response.ok) {
        throw new Error('Failed to get upload URL');
      }

      const presignedData = await response.json();

      // Upload file with progress tracking
      const result = await uploadWithPresignedUrl(presignedData, file, {
        onProgress: (progress) => {
          setProgress(progress.percent);
        },
      });

      logInfo('Upload complete', {
        key: result.key,
        url: result.url,
        size: result.size,
        contentType: result.contentType,
      });
    } catch (error) {
      logError('Upload failed', {
        error: error.message,
        file: {
          name: file.name,
          size: file.size,
          type: file.type,
        },
      });
    } finally {
      setUploading(false);
    }
  };

  return (
    <div>
      <input
        type="file"
        onChange={(e) => e.target.files?.[0] && handleUpload(e.target.files[0])}
        disabled={uploading}
      />
      {uploading && (
        <div>
          <progress value={progress} max={100} />
          <span>{progress.toFixed(1)}%</span>
        </div>
      )}
    </div>
  );
}
```

## Advanced Upload Components

### Multi-File Uploader

```typescript
'use client';

import { useState, useCallback } from 'react';
import { uploadWithPresignedUrl } from '@repo/storage/client/next';
import { logInfo, logError } from '@repo/observability/server/next';

interface UploadState {
  file: File;
  progress: number;
  status: 'pending' | 'uploading' | 'completed' | 'error';
  error?: string;
  url?: string;
}

export function MultiFileUploader() {
  const [uploads, setUploads] = useState<Map<string, UploadState>>(new Map());

  const updateUpload = useCallback((id: string, update: Partial<UploadState>) => {
    setUploads(prev => {
      const newMap = new Map(prev);
      const current = newMap.get(id);
      if (current) {
        newMap.set(id, { ...current, ...update });
      }
      return newMap;
    });
  }, []);

  const uploadFile = async (file: File) => {
    const id = `${file.name}-${Date.now()}`;

    // Add to upload queue
    setUploads(prev => new Map(prev).set(id, {
      file,
      progress: 0,
      status: 'pending',
    }));

    try {
      updateUpload(id, { status: 'uploading' });

      // Get presigned URL
      const response = await fetch('/api/upload-url', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          filename: file.name,
          contentType: file.type,
          size: file.size,
        }),
      });

      const presignedData = await response.json();

      // Upload with progress tracking
      await uploadWithPresignedUrl(presignedData, file, {
        onProgress: (progress) => {
          updateUpload(id, { progress: progress.percent });
        },
      });

      updateUpload(id, {
        status: 'completed',
        progress: 100,
        url: presignedData.url,
      });

    } catch (error) {
      updateUpload(id, {
        status: 'error',
        error: error instanceof Error ? error.message : 'Upload failed',
      });
    }
  };

  const handleFiles = (files: FileList) => {
    Array.from(files).forEach(uploadFile);
  };

  return (
    <div>
      <input
        type="file"
        multiple
        onChange={(e) => e.target.files && handleFiles(e.target.files)}
      />

      <div>
        {Array.from(uploads.entries()).map(([id, upload]) => (
          <div key={id} className="upload-item">
            <div>{upload.file.name}</div>
            <div>
              {upload.status === 'uploading' && (
                <progress value={upload.progress} max={100} />
              )}
              {upload.status === 'completed' && <span>✅ Complete</span>}
              {upload.status === 'error' && <span>❌ {upload.error}</span>}
            </div>
          </div>
        ))}
      </div>
    </div>
  );
}
```

### Drag & Drop Uploader

```typescript
'use client';

import { useState, useRef, DragEvent } from 'react';
import { uploadWithPresignedUrl } from '@repo/storage/client/next';
import { logInfo, logError } from '@repo/observability/server/next';

export function DragDropUploader() {
  const [isDragging, setIsDragging] = useState(false);
  const [uploads, setUploads] = useState<any[]>([]);
  const fileInputRef = useRef<HTMLInputElement>(null);

  const handleDragOver = (e: DragEvent) => {
    e.preventDefault();
    setIsDragging(true);
  };

  const handleDragLeave = (e: DragEvent) => {
    e.preventDefault();
    setIsDragging(false);
  };

  const handleDrop = (e: DragEvent) => {
    e.preventDefault();
    setIsDragging(false);

    const files = Array.from(e.dataTransfer.files);
    uploadFiles(files);
  };

  const uploadFiles = async (files: File[]) => {
    for (const file of files) {
      const uploadId = `${file.name}-${Date.now()}`;

      // Add to uploads list
      setUploads(prev => [...prev, {
        id: uploadId,
        name: file.name,
        size: file.size,
        progress: 0,
        status: 'uploading',
      }]);

      try {
        // Get presigned URL
        const response = await fetch('/api/upload-url', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            filename: file.name,
            contentType: file.type,
            size: file.size,
          }),
        });

        const presignedData = await response.json();

        // Upload file
        await uploadWithPresignedUrl(presignedData, file, {
          onProgress: (progress) => {
            setUploads(prev => prev.map(upload =>
              upload.id === uploadId
                ? { ...upload, progress: progress.percent }
                : upload
            ));
          },
        });

        // Mark as completed
        setUploads(prev => prev.map(upload =>
          upload.id === uploadId
            ? { ...upload, status: 'completed', progress: 100 }
            : upload
        ));

      } catch (error) {
        setUploads(prev => prev.map(upload =>
          upload.id === uploadId
            ? { ...upload, status: 'error', error: error.message }
            : upload
        ));
      }
    }
  };

  return (
    <div>
      <div
        className={`drop-zone ${isDragging ? 'dragging' : ''}`}
        onDragOver={handleDragOver}
        onDragLeave={handleDragLeave}
        onDrop={handleDrop}
        onClick={() => fileInputRef.current?.click()}
        style={{
          border: '2px dashed #ccc',
          borderRadius: '8px',
          padding: '40px',
          textAlign: 'center',
          cursor: 'pointer',
          backgroundColor: isDragging ? '#f0f0f0' : 'transparent',
        }}
      >
        <input
          ref={fileInputRef}
          type="file"
          multiple
          style={{ display: 'none' }}
          onChange={(e) => e.target.files && uploadFiles(Array.from(e.target.files))}
        />
        <p>Drop files here or click to select</p>
      </div>

      {uploads.length > 0 && (
        <div>
          <h3>Uploads</h3>
          {uploads.map(upload => (
            <div key={upload.id} className="upload-item">
              <div>{upload.name} ({(upload.size / 1024 / 1024).toFixed(2)} MB)</div>
              <div>
                {upload.status === 'uploading' && (
                  <progress value={upload.progress} max={100} />
                )}
                <span>{upload.status}</span>
              </div>
            </div>
          ))}
        </div>
      )}
    </div>
  );
}
```

## Large File Uploads

### Multipart Upload Coordination

```typescript
'use client';

import { useState } from 'react';
import {
  ClientMultipartUpload,
  splitFileIntoChunks
} from '@repo/storage/client/next';
import { logInfo, logError } from '@repo/observability/server/next';

export function LargeFileUploader() {
  const [uploading, setUploading] = useState(false);
  const [progress, setProgress] = useState(0);
  const [currentPart, setCurrentPart] = useState(0);
  const [totalParts, setTotalParts] = useState(0);

  const uploadLargeFile = async (file: File) => {
    if (file.size < 100 * 1024 * 1024) {
      // Use simple upload for files < 100MB
      return uploadSimple(file);
    }

    setUploading(true);
    setProgress(0);

    try {
      // Initialize multipart upload
      const initResponse = await fetch('/api/multipart/init', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          filename: file.name,
          contentType: file.type,
          size: file.size,
        }),
      });

      const { uploadId, key } = await initResponse.json();
      const upload = new ClientMultipartUpload(uploadId, key);

      // Split file into chunks
      const chunkSize = 50 * 1024 * 1024; // 50MB chunks
      const chunks = splitFileIntoChunks(file, chunkSize);
      const totalChunks = Math.ceil(file.size / chunkSize);
      setTotalParts(totalChunks);

      let completedParts = 0;

      // Upload each chunk
      for await (const { chunk, partNumber } of chunks) {
        setCurrentPart(partNumber);

        // Get presigned URL for this part
        const partResponse = await fetch('/api/multipart/part-url', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            uploadId,
            key,
            partNumber,
          }),
        });

        const { url } = await partResponse.json();

        // Upload part
        await upload.uploadPart(partNumber, url, chunk);

        completedParts++;
        setProgress((completedParts / totalChunks) * 100);
      }

      // Complete multipart upload
      const completeResponse = await fetch('/api/multipart/complete', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          uploadId,
          key,
          parts: upload.getParts(),
        }),
      });

      const result = await completeResponse.json();
      logInfo('Multipart upload completed', {
        uploadId: uploadId,
        key: key,
        parts: result.parts,
      });

    } catch (error) {
      logError('Multipart upload failed', {
        error: error.message,
        file: {
          name: file.name,
          size: file.size,
          type: file.type,
        },
      });
      // Optionally abort the multipart upload
    } finally {
      setUploading(false);
    }
  };

  return (
    <div>
      <input
        type="file"
        onChange={(e) => e.target.files?.[0] && uploadLargeFile(e.target.files[0])}
        disabled={uploading}
      />

      {uploading && (
        <div>
          <div>Uploading part {currentPart} of {totalParts}</div>
          <progress value={progress} max={100} />
          <div>{progress.toFixed(1)}% complete</div>
        </div>
      )}
    </div>
  );
}
```

```typescript
}
```

```typescript
import { useState } from 'react';

import { logInfo, logError } from '@repo/observability/server/next';

export function ImageUploader() {
  const [preview, setPreview] = useState<string | null>(null);
  const [uploading, setUploading] = useState(false);
  const [uploadedUrl, setUploadedUrl] = useState<string | null>(null);

  const handleFileSelect = (file: File) => {
    // Validate file type
    if (!file.type.startsWith('image/')) {
      alert('Please select an image file');
      return;

      return;

    const reader = new FileReader();
    reader.onload = (e) => {
      setPreview(e.target?.result as string);

    reader.readAsDataURL(file);

    uploadImage(file);
  };

    setUploading(true);

    try {
      // Get upload URL for Cloudflare Images
      const response = await fetch('/api/images/upload-url', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          filename: file.name,
          contentType: file.type,
        }),
      });

      const { uploadURL, imageId } = await response.json();

      // Upload to Cloudflare Images
      const formData = new FormData();
      formData.append('file', file);

        method: 'POST',
        body: formData,
```

## Error Handling

### Comprehensive Error Handling

```typescript
'use client';

import { useState } from 'react';
import { uploadWithPresignedUrl } from '@repo/storage/client/next';
import { logInfo, logError } from '@repo/observability/server/next';

interface UploadError {
  type: 'network' | 'server' | 'client' | 'unknown';
  message: string;
  retryable: boolean;
}

export function RobustUploader() {
  const [error, setError] = useState<UploadError | null>(null);
  const [retryCount, setRetryCount] = useState(0);
  const maxRetries = 3;

  const classifyError = (error: any): UploadError => {
    if (error.name === 'NetworkError' || error.message.includes('fetch')) {
      return {
        type: 'network',
        message: 'Network connection failed',
        retryable: true,
      };
    }

    if (error.status >= 500) {
      return {
        type: 'server',
        message: 'Server error occurred',
        retryable: true,
      };
    }

    if (error.status === 413) {
      return {
        type: 'client',
        message: 'File too large',
        retryable: false,
      };
    }

    if (error.status === 415) {
      return {
        type: 'client',
        message: 'Unsupported file type',
        retryable: false,
      };
    }

    return {
      type: 'unknown',
      message: error.message || 'Upload failed',
      retryable: true,
    };
  };

  const uploadWithRetry = async (file: File): Promise<void> => {
    for (let attempt = 1; attempt <= maxRetries; attempt++) {
      try {
        setError(null);

        // Get presigned URL
        const response = await fetch('/api/upload-url', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            filename: file.name,
            contentType: file.type,
            size: file.size,
          }),
        });

        if (!response.ok) {
          throw new Error(`HTTP ${response.status}: ${response.statusText}`);
        }

        const presignedData = await response.json();

        // Upload file
        await uploadWithPresignedUrl(presignedData, file, {
          onProgress: (progress) => {
            logInfo('Upload progress', {
              percent: Math.round((progress.loaded / progress.total) * 100),
              loaded: progress.loaded,
              total: progress.total,
              speed: progress.speed,
            });
          },
        });

        // Success
        setRetryCount(0);
        return;

      } catch (error) {
        const uploadError = classifyError(error);
        setError(uploadError);
        setRetryCount(attempt);

        if (!uploadError.retryable || attempt === maxRetries) {
          logError('Upload failed after retries', {
            error: uploadError.message,
            file: {
              name: file.name,
              size: file.size,
              type: file.type,
            },
            attempt: attempt,
            maxRetries: maxRetries,
          });
          throw uploadError;
        }

        // Wait before retry (exponential backoff)
        const delay = Math.pow(2, attempt - 1) * 1000;
        await new Promise(resolve => setTimeout(resolve, delay));
      }
    }
  };

  const handleUpload = async (file: File) => {
    try {
      await uploadWithRetry(file);
      logInfo('Upload successful', {
        file: {
          name: file.name,
          size: file.size,
          type: file.type,
        },
      });
    } catch (error) {
      logError('Upload failed after retries', {
        error: error.message,
        file: {
          name: file.name,
          size: file.size,
          type: file.type,
        },
      });
    }
  };

  return (
    <div>
      <input
        type="file"
        onChange={(e) => e.target.files?.[0] && handleUpload(e.target.files[0])}
      />

      {error && (
        <div className={`error ${error.type}`}>
          <p>{error.message}</p>
          {error.retryable && retryCount < maxRetries && (
            <p>Retrying... (attempt {retryCount}/{maxRetries})</p>
          )}
          {!error.retryable && (
            <p>Please check your file and try again.</p>
          )}
        </div>
      )}
    </div>
  );
}
```

### Secure Upload Implementation

```typescript
// Client-side validation
const validateFile = (file: File): { valid: boolean; error?: string } => {
  // Check file size
  const maxSize = 100 * 1024 * 1024; // 100MB
  if (file.size > maxSize) {
    return { valid: false, error: "File too large" };
  }

  // Check file type
  const allowedTypes = ["image/jpeg", "image/png", "application/pdf"];
  if (!allowedTypes.includes(file.type)) {
    return { valid: false, error: "File type not allowed" };
  }

  // Check file name
  if (!/^[a-zA-Z0-9._-]+$/.test(file.name)) {
    return { valid: false, error: "Invalid file name" };
  }

  return { valid: true };
};

// Secure upload function
const secureUpload = async (file: File) => {
  // Validate file
  const validation = validateFile(file);
  if (!validation.valid) {
    throw new Error(validation.error);
  }

  // Get presigned URL with validation
  const response = await fetch("/api/upload-url", {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      Authorization: `Bearer ${await getAuthToken()}` // Include auth
    },
    body: JSON.stringify({
      filename: file.name,
      contentType: file.type,
      size: file.size,
      checksum: await generateChecksum(file) // File integrity
    })
  });

  if (!response.ok) {
    throw new Error("Failed to get upload URL");
  }

  const presignedData = await response.json();

  // Upload with validation
  return uploadWithPresignedUrl(presignedData, file, {
    validateResponse: true,
    timeout: 60000 // 60 second timeout
  });
};

// Checksum generation
const generateChecksum = async (file: File): Promise<string> => {
  const buffer = await file.arrayBuffer();
  const hashBuffer = await crypto.subtle.digest("SHA-256", buffer);
  const hashArray = Array.from(new Uint8Array(hashBuffer));
  return hashArray.map((b) => b.toString(16).padStart(2, "0")).join("");
};
```

## React Hooks

### Custom Upload Hook

```typescript
'use client';

import { useState, useCallback } from 'react';
import { uploadWithPresignedUrl } from '@repo/storage/client/next';
import { logInfo, logError } from '@repo/observability/server/next';

interface UseUploadOptions {
  onSuccess?: (url: string) => void;
  onError?: (error: Error) => void;
  maxRetries?: number;
}

export function useUpload(options: UseUploadOptions = {}) {
  const [uploading, setUploading] = useState(false);
  const [progress, setProgress] = useState(0);
  const [error, setError] = useState<Error | null>(null);

  const upload = useCallback(async (file: File) => {
    setUploading(true);
    setProgress(0);
    setError(null);

    try {
      // Get presigned URL
      const response = await fetch('/api/upload-url', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          filename: file.name,
          contentType: file.type,
          size: file.size,
        }),
      });

      const presignedData = await response.json();

      // Upload file
      await uploadWithPresignedUrl(presignedData, file, {
        onProgress: (prog) => setProgress(prog.percent),
      });

      options.onSuccess?.(presignedData.url);
    } catch (err) {
      const error = err instanceof Error ? err : new Error('Upload failed');
      setError(error);
      options.onError?.(error);
    } finally {
      setUploading(false);
    }
  }, [options]);

  const reset = useCallback(() => {
    setUploading(false);
    setProgress(0);
    setError(null);
  }, []);

  return { upload, uploading, progress, error, reset };
}

// Usage
export function UploadButton() {
  const { upload, uploading, progress, error } = useUpload({
    onSuccess: (url) => logInfo('Uploaded', { url: url }),
    onError: (error) => logError('Error', { error: error.message }),
  });

  return (
    <div>
      <input
        type="file"
        onChange={(e) => e.target.files?.[0] && upload(e.target.files[0])}
        disabled={uploading}
      />
      {uploading && <progress value={progress} max={100} />}
      {error && <p>Error: {error.message}</p>}
    </div>
  );
}
```

```typescript
// Chunked upload for better performance
const optimizedUpload = async (file: File) => {
  const chunkSize =
    file.size > 100 * 1024 * 1024
      ? 50 * 1024 * 1024 // 50MB chunks for large files
      : 10 * 1024 * 1024; // 10MB chunks for smaller files

  if (file.size > chunkSize) {
    return uploadAsMultipart(file, chunkSize);
  } else {
    return uploadAsSimple(file);
  }
};

// Parallel uploads for multiple files
const uploadMultipleFiles = async (files: File[]) => {
  const maxConcurrent = 3; // Limit concurrent uploads
  const semaphore = new Array(maxConcurrent).fill(Promise.resolve());

  return Promise.all(
    files.map(async (file, index) => {
      await semaphore[index % maxConcurrent];

      const uploadPromise = optimizedUpload(file);
      semaphore[index % maxConcurrent] = uploadPromise;

      return uploadPromise;
    })
  );
};

// Progressive loading for large files
const uploadWithProgressiveLoading = async (file: File) => {
  // Upload thumbnail/preview first
  const thumbnail = await generateThumbnail(file);
  const thumbnailResult = await uploadWithPresignedUrl(
    await getPresignedUrl(`thumb-${file.name}`),
    thumbnail
  );

  // Then upload full file in background
  const fullResult = await uploadWithPresignedUrl(
    await getPresignedUrl(file.name),
    file,
    {
      priority: "background"
    }
  );

  return { thumbnail: thumbnailResult, full: fullResult };
};
```
