HTML Integration

<!-- Include the standalone bundle -->
<script src="./embeddable-standalone.js"></script>

<!-- Create container -->
<div id="my-editor"></div>

<script>
  // Initialize editor
  const editor = new NotionEditor.EmbeddableNotionEditor({
    container: document.getElementById('my-editor'),
    placeholder: 'Start writing...',
    content: '<p>Hello world!</p>',
    onChange: (html) => console.log('Content:', html),
    theme: {
      backgroundColor: '#ffffff',
      textColor: '#1f2937',
      borderColor: '#e5e7eb',
      focusColor: '#3b82f6'
    }
  });
</script>

React Integration

import { EmbeddableNotionEditor } from '@repo/editor/react';

function MyComponent() {
  return (
    <EmbeddableNotionEditor
      placeholder="Start writing..."
      onChange={(html) => console.log(html)}
      features={{
        formatting: true,
        lists: true,
        links: true
      }}
      theme={{
        backgroundColor: '#ffffff',
        focusColor: '#3b82f6'
      }}
    />
  );
}

Next.js Integration

import dynamic from 'next/dynamic';

const NotionEditor = dynamic(
  () => import('@repo/editor/nextjs').then(mod => mod.EmbeddableNotionEditor),
  { ssr: false }
);

export default function Page() {
  return (
    <NotionEditor
      placeholder="Start writing..."
      onChange={(html) => console.log(html)}
    />
  );
}
